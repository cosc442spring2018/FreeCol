478         final ClientOptions co = freeColClient.getClientOptions();
479         if (co.getBoolean(ClientOptions.AUTOSAVE_DELETE)) {
480             FreeColServer.removeAutosaves(co.getText(ClientOptions.AUTO_SAVE_PREFIX));
481         }
482         freeColClient.getPreGameController().setReady(true);
483         if (skip) {
484             freeColClient.getPreGameController().requestLaunch();
485         } else {
486             gui.showStartGamePanel(freeColClient.getGame(),
487                                    freeColClient.getMyPlayer(), true);
488         }
489         return true;
490     }

492     /**
493      * Loads and starts a game from the given file.
494      *
495      * @param file The saved game.
496      * @param userMsg An optional message key to be displayed early.
497      * @return True if the game starts successully.
498      */
499     public boolean startSavedGame(File file, final String userMsg) {
500         freeColClient.setMapEditor(false);
501
502         class ErrorJob implements Runnable {
503             private final String message;
504             private final StringTemplate template;
505             
506             ErrorJob(String message) {
507                 this.message = message;
508                 this.template = null;
509             }
510
511             ErrorJob(StringTemplate template) {
512                 this.message = null;
513                 this.template = template;
514             }
515             
516             @Override
517             public void run() {
518                 gui.closeMenus();
519                 if (this.template != null) {
520                     gui.showErrorMessage(template);
521                 } else {
522                     gui.showErrorMessage(message);
523                 }
524             }
525         }
526
527         final ClientOptions options = freeColClient.getClientOptions();
528         final boolean defaultSinglePlayer;
529         final boolean defaultPublicServer;
530         FreeColSavegameFile fis = null;
531         try {
532             fis = new FreeColSavegameFile(file);
533         } catch (IOException ioe) {
534             SwingUtilities.invokeLater(new ErrorJob(FreeCol.badLoad(file)));
535             logger.log(Level.WARNING, "Could not open save file: "
536                 + file.getName());
537             return false;
538         }
539         options.merge(fis);
540         options.fixClientOptions();
541
542         // Get suggestions for "singlePlayer" and "publicServer"
543         // settings from the file, and update the client options if
544         // possible.
545         try (
546             FreeColXMLReader xr = fis.getFreeColXMLReader();
547         ) {
548             xr.nextTag();
549
550             String str = xr.getAttribute(FreeColServer.OWNER_TAG,
551                                          (String)null);
552             if (str != null) FreeCol.setName(str);
553
554             defaultSinglePlayer
555                 = xr.getAttribute(FreeColServer.SINGLE_PLAYER_TAG, false);
556             defaultPublicServer
557                 = xr.getAttribute(FreeColServer.PUBLIC_SERVER_TAG, false);
558
559         } catch (FileNotFoundException e) {
560             SwingUtilities.invokeLater(new ErrorJob("server.fileNotFound"));
561             logger.log(Level.WARNING, "Can not find file: " + file.getName(),
562                 e);
563             return false;
564         } catch (XMLStreamException e) {
565             logger.log(Level.WARNING, "Error reading game from: "
566                 + file.getName(), e);
567             SwingUtilities.invokeLater(new ErrorJob(FreeCol.badLoad(file)));
568             return false;
569         } catch (Exception e) {
570             SwingUtilities.invokeLater(new ErrorJob(FreeCol.badLoad(file)));
571             logger.log(Level.WARNING, "Could not load game from: "
572                 + file.getName(), e);
573             return false;
574         }
575
576         // Reload the client options saved with this game.
577         final boolean singlePlayer;
578         final String name;
579         final int port;
580         final int sgo = options.getInteger(ClientOptions.SHOW_SAVEGAME_SETTINGS);
581         boolean show = sgo == ClientOptions.SHOW_SAVEGAME_SETTINGS_ALWAYS
582             || (!defaultSinglePlayer
583                 && sgo == ClientOptions.SHOW_SAVEGAME_SETTINGS_MULTIPLAYER);
584         if (show) {
585             if (!gui.showLoadingSavegameDialog(defaultPublicServer,
586                                                defaultSinglePlayer))
587                 return false;
588             LoadingSavegameInfo lsd = gui.getLoadingSavegameInfo();
589             singlePlayer = lsd.isSinglePlayer();
590             name = lsd.getServerName();
591             port = lsd.getPort();
592         } else {
593             singlePlayer = defaultSinglePlayer;
594             name = null;
595             port = -1;
596         }
597         Messages.loadActiveModMessageBundle(options.getActiveMods(),
598                                             FreeCol.getLocale());
599         if (!unblockServer(port)) return false;
600         gui.showStatusPanel(Messages.message("status.loadingGame"));
601
602         final File theFile = file;
603         Runnable loadGameJob = () -> {
604             FreeColServer freeColServer = null;
605             StringTemplate err = null;
606             try {
607                 final FreeColSavegameFile saveGame
608                     = new FreeColSavegameFile(theFile);
609                 freeColServer = new FreeColServer(saveGame,
610                     (Specification)null, port, name);
611                 freeColClient.setFreeColServer(freeColServer);
612                 // Server might have bounced to another port.
613                 freeColClient.setSinglePlayer(singlePlayer);
614                 freeColClient.getInGameController().setGameConnected();
615                 if (login(FreeCol.getName(), freeColServer.getHost(), 
616                         freeColServer.getPort())) {
617                     SwingUtilities.invokeLater(() -> {
618                             ResourceManager.setScenarioMapping(saveGame.getResourceMapping());
619                             if (userMsg != null) {
620                                 gui.showInformationMessage(userMsg);
621                             }
622                             gui.closeStatusPanel();
623                         });
624                     return; // Success!
625                 }
626                 err = StringTemplate.key("server.couldNotLogin");
627                 logger.warning("Could not log in.");
628             } catch (FileNotFoundException e) {
629                 err = StringTemplate.key("server.fileNotFound");
630                 logger.log(Level.WARNING, "Can not find file.", e);
631             } catch (IOException e) {
632                 err = StringTemplate.key("server.initialize");
633                 logger.log(Level.WARNING, "Error starting game.", e);
634             } catch (XMLStreamException e) {
635                 err = FreeCol.badLoad(theFile);
636                 logger.log(Level.WARNING, "Stream error.", e);
637             } catch (Exception e) {
638                 err = StringTemplate.name(e.getMessage());
639                 logger.log(Level.WARNING, "FreeCol error.", e);
640             }                
641             if (err != null) {
642                 // If this is a debug run, fail hard.
643                 if (freeColClient.isHeadless()
644                     || FreeColDebugger.getDebugRunTurns() >= 0) {
645                     FreeCol.fatal(Messages.message(err));
646                 }
647                 SwingUtilities.invokeLater(() -> {
648                         gui.closeMainPanel();
649                         gui.showMainPanel(null);
650                     });
651                 SwingUtilities.invokeLater(new ErrorJob(err));
652             }
653         };
654         freeColClient.setWork(loadGameJob);
655         return true;
656     }

724      * @return A list of {@link ServerInfo} objects.

728             Connection mc = new Connection(FreeCol.META_SERVER_ADDRESS,
729                 FreeCol.META_SERVER_PORT, null,
730                 FreeCol.CLIENT_THREAD);
731         ) {
732             Element reply = null;
733             try {
734                 reply = mc.ask(DOMMessage.createMessage("getServerList"));
735             } catch (IOException e) {
736                 reply = null;
737             }
738             if (reply == null) {
739                 gui.showErrorMessage("metaServer.communicationError");
740                 logger.warning("The meta-server did not return a list.");

746                 items.add(new ServerInfo((Element)nl.item(i)));
747             }
748             return items;
749         } catch (IOException e) {
750             gui.showErrorMessage("metaServer.couldNotConnect");
751             logger.log(Level.WARNING, "Could not connect to the meta-server.", e);
752             return null;
753         }
