64     /**
65      * Handles a network message.
66      *
67      * @param conn The <code>Connection</code> the message came from.
68      * @param element The message to be processed.
69      * @return The reply.
70      */
71     @Override
72     public synchronized Element handle(Connection conn, Element element) {
73         final String tag = element.getTagName();
74         return ("disconnect".equals(tag)) 
75             ? disconnect(conn, element)
76             : ("gameState".equals(tag))
77             ? gameState(conn, element)
78             : ("getVacantPlayers".equals(tag))
79             ? getVacantPlayers(conn, element)
80             : ("login".equals(tag))
81             ? login(conn, element)
82             : unknown(tag);
83     }

131         @SuppressWarnings("unused") Element element) {

134
135         Element reply = DOMMessage.createMessage("vacantPlayers");

137         for (Player p : game.getLiveEuropeanPlayers(null)) {
138             if (!p.isREF()

148     /**
149      * Handles a "login"-request.
150      *
151      * FIXME: Do not allow more than one (human) player to connect
152      * to a single player game. This would be easy if we used a
153      * dummy connection for single player games.
154      *
155      * @param connection The <code>Connection</code> the message was
156      *     received on.
157      * @param element The <code>Element</code> (root element in a
158      *     DOM-parsed XML tree) that holds all the information.
159      * @return The reply.
160      */
161     private Element login(Connection connection, Element element) {
162         final String userName = element.getAttribute("userName");
163         final String version = element.getAttribute("version");
164
165         if (userName == null || userName.isEmpty()) {
166             return DOMMessage.createError("server.missingUserName", null);
167         } else if (version == null || version.isEmpty()) {
168             return DOMMessage.createError("server.missingVersion", null);
169         } else if (!version.equals(FreeCol.getVersion())) {
170             return DOMMessage.createError("server.wrongFreeColVersion",
171                 version + " != " + FreeCol.getVersion());
172         }
173
174         final FreeColServer freeColServer = getFreeColServer();
175         final Server server = freeColServer.getServer();
176         Game game;
177         ServerPlayer player;
178         Unit active = null;
179         boolean isCurrentPlayer = false;
180         MessageHandler mh;
181         boolean starting = freeColServer.getGameState()
182             == FreeColServer.GameState.STARTING_GAME;
183         if (starting) {
184             // Wait until the game has been created.
185             // FIXME: is this still needed?
186             int timeOut = 20000;
187             while (freeColServer.getGame() == null) {
188                 try {
189                     Thread.sleep(1000);
190                 } catch (InterruptedException e) {}
191                 if ((timeOut -= 1000) <= 0) {
192                     return DOMMessage.createError("server.timeOut", null);
193                 }
194             }
195
196             game = freeColServer.getGame();
197             if (!game.canAddNewPlayer()) {
198                 return DOMMessage.createError("server.maximumPlayers", null);
199             } else if (game.playerNameInUse(userName)) {
200                 return DOMMessage.createError("server.userNameInUse",
201                     userName + " is already in use.");
202             }
203
204             // Create and add the new player:
205             boolean admin = game.getLivePlayers(null).isEmpty();
206             player = new ServerPlayer(game, admin, game.getVacantNation(),
207                                       connection.getSocket(), connection);
208             player.setName(userName);
209             game.addPlayer(player);
210
211             // Send message to all players except to the new player.
212             // FIXME: check visibility.
213             Element add = DOMMessage.createMessage("addPlayer");
214             add.appendChild(player.toXMLElement(add.getOwnerDocument()));
215             server.sendToAll(add, connection);
216
217             // Ready now to handle pre-game messages.
218             mh = freeColServer.getPreGameInputHandler();
219
220         } else { // Restoring from existing game.
221             game = freeColServer.getGame();
222             player = (ServerPlayer)game.getPlayerByName(userName);
223             if (player == null) {
224                 StringBuilder sb = new StringBuilder("Player \"");
225                 sb.append(userName).append("\" is not present in the game.")
226                     .append("\n  Known players = ( ");
227                 for (Player p : game.getLiveEuropeanPlayers(null)) {
228                     sb.append(p.getName()).append(" ");
229                 }
230                 sb.append(")");
231                 return DOMMessage.createError("server.userNameNotPresent",
232                                               sb.toString());
233             } else if (player.isConnected() && !player.isAI()) {
234                 return DOMMessage.createError("server.userNameInUse",
235                     userName + " is already in use.");
236             }
237             player.setConnection(connection);
238             player.setConnected(true);
239
240             if (player.isAI()) {
241                 player.setAI(false);
242                 server.sendToAll(DOMMessage.createMessage("setAI",
243                         "player", player.getId(),
244                         "ai", Boolean.toString(false)));
245             }
246
247             // If this player is the first to reconnect, it is the
248             // current player.
249             isCurrentPlayer = game.getCurrentPlayer() == null;
250             if (isCurrentPlayer) {
251                 game.setCurrentPlayer(player);
252                 active = freeColServer.getActiveUnit();
253             }
254
255             // Go straight into the game.
256             mh = freeColServer.getInGameInputHandler();
257         }
258
259         connection.setMessageHandler(mh);
260         server.addConnection(connection);
261         freeColServer.updateMetaServer();
262         return new LoginMessage(player, userName, version, !starting,
263                                 freeColServer.getSinglePlayer(),
264                                 isCurrentPlayer, active,
265                                 game).toXMLElement();
266     }
