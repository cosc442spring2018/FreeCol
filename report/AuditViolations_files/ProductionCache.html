130     private List<Entry> createEntries(GoodsType goodsType) {
131         // FIXME: OO/generic
132         List<Entry> result = new ArrayList<>();
133         if (goodsType.isFarmed()) {
134             for (ColonyTile colonyTile : colonyTiles) {
135                 Tile tile = colonyTile.getWorkTile();
136                 if (tile.getPotentialProduction(goodsType, null) > 0
137                     || (tile.hasResource()
138                         && !tile.getTileItemContainer().getResource().getType()
139                         .getModifiers(goodsType.getId()).isEmpty())) {
140                     for (Unit unit : units) {
141                         result.add(new Entry(goodsType, colonyTile, unit));
142                     }
143                 }
144             }
145         } else {
146             for (WorkLocation wl : colony.getWorkLocationsForProducing(goodsType)) {
147                 if (!(wl instanceof Building)) continue;
148                 Building building = (Building)wl;
149                 if (building.getType().getWorkPlaces() > 0) {
150                     for (Unit unit : units) {
151                         result.add(new Entry(goodsType, building, unit));
152                     }
153                 }
154             }
155         }
156         Collections.sort(result, defaultComparator);
157         entries.put(goodsType, result);
